  def chars(content, options={})
    # defaults
    attributes = options[:attributes] || {}

    value = self.tag('value', content)
    chars = self.tag("chars", value, attributes )

    cell = options[:cell] || {}
    row = options[:row] || {}
    self.row_cell(chars, cell, row)
  end
